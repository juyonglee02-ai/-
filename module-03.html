이론

3-1. 토큰이란? (15분)

배포 서버는 아무나 파일을 받아가면 안 됩니다. 미리 발급·등록된 토큰과 요청 액션(update/migration/ntp)이 일치할 때만 서비스합니다.

클라이언트 → 서버:  UPD-1-1.0.0-A1B2C3D4|update
서버 검증: tokens.conf 또는 접두사 규칙(UPD→update)
서버 응답: OK|update started → FILE|... → OK|update complete
  • Flask: 토큰 발급·이력·통계 (정책)
  • C 서버: 토큰 검증·전송 (실행)
  • tokens.conf: C 서버가 읽는 유효 토큰 목록

★ 실습 3: 토큰 연동 (20분)

  1. Flask에서 발급한 토큰을 config/tokens.conf에 한 줄 추가
  2. 형식: 토큰값=update|5 (5회까지 사용 가능)
  3. 마이그레이션 토큰 형식을 GUI에서 새로 만들고 (MIG-LAB-{random}) 발급
  4. packages/config.ini의 version을 1.0.2로 수정 (다음 실습용)
Flask DB의 토큰과 tokens.conf는 별도입니다. 실무에서는 API로 자동 동기화할 수 있습니다(확장 과제).

/*
 * config.c — 설정 파일·토큰 파일 읽기
 * deploy_targets.conf : 포트, 패키지 경로, 동시 접속 수 등
 * tokens.conf         : 허용된 배포 토큰 목록

#include "config.h"
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <ctype.h>

/* 여러 스레드가 동시에 토큰을 검증할 때 값이 꼬이지 않도록 잠금(뮤텍스) 사용 */
#ifdef _WIN32
  #include <windows.h>
  static CRITICAL_SECTION g_token_cs;
  #define TOKEN_LOCK() EnterCriticalSection(&g_token_cs)
  #define TOKEN_UNLOCK() LeaveCriticalSection(&g_token_cs)
#else
  #include <pthread.h>
  static pthread_mutex_t g_token_mtx = PTHREAD_MUTEX_INITIALIZER;
  #define TOKEN_LOCK() pthread_mutex_lock(&g_token_mtx)
  #define TOKEN_UNLOCK() pthread_mutex_unlock(&g_token_mtx)
#endif

void token_lock_init(void) {
#ifdef _WIN32
    InitializeCriticalSection(&g_token_cs);
#endif
}

void token_lock_shutdown(void) {
#ifdef _WIN32
    DeleteCriticalSection(&g_token_cs);
#endif
}

/* 문자열 앞뒤 공백 제거 (설정 파일 파싱용) */
static void trim(char *s) {
    char *p = s;
    while (*p && isspace((unsigned char)*p)) p++;
    if (p != s) memmove(s, p, strlen(p) + 1);
    size_t n = strlen(s);
    while (n > 0 && isspace((unsigned char)s[n - 1])) s[--n] = '\0';
}

/* update_file_1=xxx 형식으로 파일 이름을 배열에 추가 */
static void add_file(char files[][MAX_PATH_LEN], int *count, const char *path) {
    if (*count >= MAX_FILES) return;
    strncpy(files[*count], path, MAX_PATH_LEN - 1);
    files[*count][MAX_PATH_LEN - 1] = '\0';
    (*count)++;
}

/*
 * deploy_targets.conf 를 읽어 deploy_config_t 구조체를 채움
 * 반환: 0=성공, -1=파일 없음
 */
int load_deploy_config(const char *path, deploy_config_t *cfg) {
    FILE *fp;
    char line[MAX_LINE];
    char key[64], value[MAX_PATH_LEN];

    memset(cfg, 0, sizeof(*cfg));
    /* 파일에 없는 항목은 아래 기본값 사용 */
    cfg->server_port = DEFAULT_PORT;
    strcpy(cfg->package_dir, "../packages");
    strcpy(cfg->client_install_dir, "./installed");
    cfg->worker_threads = DEFAULT_WORKER_THREADS;
    cfg->listen_backlog = DEFAULT_LISTEN_BACKLOG;
    cfg->max_queue_size = DEFAULT_MAX_QUEUE;
    cfg->max_concurrent = DEFAULT_MAX_CONCURRENT;
    cfg->package_cache_enabled = 1;
    strcpy(cfg->ntp_server, "pool.ntp.org");
    cfg->ntp_sync_on_startup = 1;
    cfg->ntp_sync_interval_min = 30;
    cfg->ntp_sync_clients = 1;

    fp = fopen(path, "r");
    if (!fp) return -1;

    while (fgets(line, sizeof(line), fp)) {
        trim(line);
        if (line[0] == '#' || line[0] == '\0') continue;  /* 주석·빈 줄 건너뜀 */
        if (sscanf(line, " %63[^=] = %259[^\n]", key, value) < 2) continue;
        trim(key);
        trim(value);

        if (strcmp(key, "server_port") == 0)
            cfg->server_port = atoi(value);
        else if (strcmp(key, "package_dir") == 0)
            strncpy(cfg->package_dir, value, MAX_PATH_LEN - 1);
        else if (strcmp(key, "client_install_dir") == 0)
            strncpy(cfg->client_install_dir, value, MAX_PATH_LEN - 1);
        else if (strcmp(key, "flask_report_url") == 0)
            strncpy(cfg->flask_report_url, value, MAX_PATH_LEN - 1);
        else if (strcmp(key, "worker_threads") == 0)
            cfg->worker_threads = atoi(value);
        else if (strcmp(key, "listen_backlog") == 0)
            cfg->listen_backlog = atoi(value);
        else if (strcmp(key, "max_queue_size") == 0)
            cfg->max_queue_size = atoi(value);
        else if (strcmp(key, "max_concurrent") == 0)
            cfg->max_concurrent = atoi(value);
        else if (strcmp(key, "package_cache_enabled") == 0)
            cfg->package_cache_enabled = atoi(value);
        else if (strcmp(key, "ntp_server") == 0)
            strncpy(cfg->ntp_server, value, MAX_PATH_LEN - 1);
        else if (strcmp(key, "ntp_sync_on_startup") == 0)
            cfg->ntp_sync_on_startup = atoi(value);
        else if (strcmp(key, "ntp_sync_interval_min") == 0)
            cfg->ntp_sync_interval_min = atoi(value);
        else if (strcmp(key, "ntp_sync_clients") == 0)
            cfg->ntp_sync_clients = atoi(value);
        else if (strncmp(key, "update_file_", 12) == 0)
            add_file(cfg->update_files, &cfg->update_file_count, value);
        else if (strncmp(key, "migration_file_", 15) == 0)
            add_file(cfg->migration_files, &cfg->migration_file_count, value);
    }
    fclose(fp);
    return 0;
}

/*
 * tokens.conf 읽기
 * 한 줄 형식: 토큰문자열=update|최대사용횟수
 * 반환: 읽은 토큰 개수
 */
int load_tokens(const char *path, token_entry_t *tokens, int max_tokens) {
    FILE *fp;
    char line[MAX_LINE];
    int count = 0;
    char token[MAX_TOKEN_LEN];
    char action_str[32];
    int max_uses;

    fp = fopen(path, "r");
    if (!fp) return 0;

    while (fgets(line, sizeof(line), fp) && count < max_tokens) {
        trim(line);
        if (line[0] == '#' || line[0] == '\0') continue;
        {
            char *eq = strchr(line, '=');
            char *pipe;
            if (!eq) continue;
            *eq = '\0';
            strncpy(token, line, MAX_TOKEN_LEN - 1);
            pipe = strchr(eq + 1, '|');
            if (pipe) {
                *pipe = '\0';
                max_uses = atoi(pipe + 1);
            } else {
                max_uses = 1;
            }
            strncpy(action_str, eq + 1, sizeof(action_str) - 1);
        }
        trim(token);
        trim(action_str);

        strncpy(tokens[count].token, token, MAX_TOKEN_LEN - 1);
        if (strcmp(action_str, "update") == 0)
            tokens[count].action = ACTION_UPDATE;
        else if (strcmp(action_str, "migration") == 0)
            tokens[count].action = ACTION_MIGRATION;
        else if (strcmp(action_str, "ntp") == 0)
            tokens[count].action = ACTION_NTP;
        else
            tokens[count].action = ACTION_UNKNOWN;
        tokens[count].max_uses = max_uses > 0 ? max_uses : 1;
        tokens[count].used = 0;  /* 아직 사용한 횟수 */
        count++;
    }
    fclose(fp);
    return count;
}

/* 토큰 문자열 앞부분(UPD, MIG, NTP)으로 작업 종류 추측 — 개발·데모용 */
deploy_action_t infer_action_from_token(const char *token) {
    if (strncmp(token, "UPD", 3) == 0 || strncmp(token, "UPDATE", 6) == 0)
        return ACTION_UPDATE;
    if (strncmp(token, "MIG", 3) == 0 || strncmp(token, "MIGRATION", 9) == 0)
        return ACTION_MIGRATION;
    if (strncmp(token, "NTP", 3) == 0)
        return ACTION_NTP;
    return ACTION_UNKNOWN;
}

/*
 * 클라이언트가 보낸 토큰이 유효한지 검사
 * - tokens.conf 에 등록되어 있는지
 * - 요청한 action(update 등)과 맞는지
 * - 사용 횟수가 남았는지
 * 반환: 1=허용, 0=거부
 */
int validate_token(const char *token, deploy_action_t requested,
                   token_entry_t *tokens, int token_count) {
    int i, ok = 0;
    deploy_action_t inferred;

    TOKEN_LOCK();
    for (i = 0; i < token_count; i++) {
        if (strcmp(tokens[i].token, token) == 0) {
            if (tokens[i].used >= tokens[i].max_uses) {
                TOKEN_UNLOCK();
                return 0;  /* 사용 횟수 초과 */
            }
            if (tokens[i].action != ACTION_UNKNOWN && tokens[i].action != requested) {
                TOKEN_UNLOCK();
                return 0;  /* 예: update 토큰으로 migration 요청 */
            }
            tokens[i].used++;
            TOKEN_UNLOCK();
            return 1;
        }
    }
    /* 목록에 없으면 접두사 규칙으로 허용 (실습용 DEMO 토큰 등) */
    inferred = infer_action_from_token(token);
    ok = (inferred == requested && inferred != ACTION_UNKNOWN);
    TOKEN_UNLOCK();
    return ok;
}
