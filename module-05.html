/*
 * thread_pool.c — 작업 큐 + 워커 스레드
 * 배포 서버가 클라이언트마다 스레드를 새로 만들지 않고,
 * 미리 만든 스레드 풀에서 꺼내 쓰면 1000대 접속에도 버틸 수 있습니다.
 */
#include "thread_pool.h"
#include <stdlib.h>
#include <string.h>

#ifdef _WIN32
  #ifndef WIN32_LEAN_AND_MEAN
  #define WIN32_LEAN_AND_MEAN
  #endif
  #include <windows.h>
#else
  #include <pthread.h>
#endif

typedef struct {
    void (*fn)(void *);
    void *arg;
} job_t;

struct thread_pool {
#ifdef _WIN32
    HANDLE *threads;
    HANDLE sem_jobs;
    HANDLE sem_slots;
    CRITICAL_SECTION mutex;
#else
    pthread_t *threads;
    pthread_mutex_t mutex;
    pthread_cond_t cond_nonempty;
    pthread_cond_t cond_nonfull;
#endif
    int num_threads;
    job_t *queue;
    int queue_capacity;
    int head, tail, count;
    int shutdown;
};

#ifdef _WIN32
static DWORD WINAPI worker_loop_win(LPVOID arg) {
    thread_pool_t *pool = (thread_pool_t *)arg;
    job_t job;
    for (;;) {
        if (WaitForSingleObject(pool->sem_jobs, INFINITE) != WAIT_OBJECT_0)
            break;
        EnterCriticalSection(&pool->mutex);
        if (pool->shutdown && pool->count == 0) {
            LeaveCriticalSection(&pool->mutex);
            break;
        }
        job = pool->queue[pool->head];
        pool->head = (pool->head + 1) % pool->queue_capacity;
        pool->count--;
        ReleaseSemaphore(pool->sem_slots, 1, NULL);
        LeaveCriticalSection(&pool->mutex);
        if (job.fn) job.fn(job.arg);
    }
    return 0;
}
#else
static void *worker_loop(void *arg) {
    thread_pool_t *pool = (thread_pool_t *)arg;
    job_t job;
    for (;;) {
        pthread_mutex_lock(&pool->mutex);
        while (pool->count == 0 && !pool->shutdown)
            pthread_cond_wait(&pool->cond_nonempty, &pool->mutex);
        if (pool->shutdown && pool->count == 0) {
            pthread_mutex_unlock(&pool->mutex);
            break;
        }
        job = pool->queue[pool->head];
        pool->head = (pool->head + 1) % pool->queue_capacity;
        pool->count--;
        pthread_cond_signal(&pool->cond_nonfull);
        pthread_mutex_unlock(&pool->mutex);
        if (job.fn) job.fn(job.arg);
    }
    return NULL;
}
#endif

thread_pool_t *thread_pool_create(int num_threads, int queue_capacity) {
    thread_pool_t *pool;
    int i;
    if (num_threads < 1) num_threads = 1;
    if (queue_capacity < 16) queue_capacity = 16;

    pool = (thread_pool_t *)calloc(1, sizeof(thread_pool_t));
    if (!pool) return NULL;
    pool->num_threads = num_threads;
    pool->queue_capacity = queue_capacity;
    pool->queue = (job_t *)calloc((size_t)queue_capacity, sizeof(job_t));
#ifdef _WIN32
    pool->threads = (HANDLE *)calloc((size_t)num_threads, sizeof(HANDLE));
    pool->sem_jobs = CreateSemaphore(NULL, 0, (LONG)queue_capacity + num_threads, NULL);
    pool->sem_slots = CreateSemaphore(NULL, (LONG)queue_capacity, (LONG)queue_capacity, NULL);
#else
    pool->threads = (pthread_t *)calloc((size_t)num_threads, sizeof(pthread_t));
    pthread_mutex_init(&pool->mutex, NULL);
    pthread_cond_init(&pool->cond_nonempty, NULL);
    pthread_cond_init(&pool->cond_nonfull, NULL);
#endif
    if (!pool->queue || !pool->threads
#ifdef _WIN32
        || !pool->sem_jobs || !pool->sem_slots
#endif
    ) {
        thread_pool_destroy(pool);
        return NULL;
    }
#ifdef _WIN32
    InitializeCriticalSection(&pool->mutex);
    for (i = 0; i < num_threads; i++)
        pool->threads[i] = CreateThread(NULL, 0, worker_loop_win, pool, 0, NULL);
#else
    for (i = 0; i < num_threads; i++)
        pthread_create(&pool->threads[i], NULL, worker_loop, pool);
#endif
    return pool;
}

int thread_pool_submit(thread_pool_t *pool, void (*task)(void *arg), void *arg) {
    if (!pool || !task || pool->shutdown) return -1;
#ifdef _WIN32
    if (WaitForSingleObject(pool->sem_slots, INFINITE) != WAIT_OBJECT_0)
        return -1;
    EnterCriticalSection(&pool->mutex);
    pool->queue[pool->tail].fn = task;
    pool->queue[pool->tail].arg = arg;
    pool->tail = (pool->tail + 1) % pool->queue_capacity;
    pool->count++;
    LeaveCriticalSection(&pool->mutex);
    ReleaseSemaphore(pool->sem_jobs, 1, NULL);
#else
    pthread_mutex_lock(&pool->mutex);
    while (pool->count >= pool->queue_capacity && !pool->shutdown)
        pthread_cond_wait(&pool->cond_nonfull, &pool->mutex);
    if (pool->shutdown) {
        pthread_mutex_unlock(&pool->mutex);
        return -1;
    }
    pool->queue[pool->tail].fn = task;
    pool->queue[pool->tail].arg = arg;
    pool->tail = (pool->tail + 1) % pool->queue_capacity;
    pool->count++;
    pthread_cond_signal(&pool->cond_nonempty);
    pthread_mutex_unlock(&pool->mutex);
#endif
    return 0;
}

int thread_pool_queue_size(thread_pool_t *pool) {
    int n;
    if (!pool) return 0;
#ifdef _WIN32
    EnterCriticalSection(&pool->mutex);
    n = pool->count;
    LeaveCriticalSection(&pool->mutex);
#else
    pthread_mutex_lock(&pool->mutex);
    n = pool->count;
    pthread_mutex_unlock(&pool->mutex);
#endif
    return n;
}

void thread_pool_destroy(thread_pool_t *pool) {
    int i;
    if (!pool) return;
#ifdef _WIN32
    EnterCriticalSection(&pool->mutex);
    pool->shutdown = 1;
    LeaveCriticalSection(&pool->mutex);
    for (i = 0; i < pool->num_threads; i++)
        ReleaseSemaphore(pool->sem_jobs, 1, NULL);
    for (i = 0; i < pool->num_threads; i++) {
        WaitForSingleObject(pool->threads[i], INFINITE);
        CloseHandle(pool->threads[i]);
    }
    CloseHandle(pool->sem_jobs);
    CloseHandle(pool->sem_slots);
    DeleteCriticalSection(&pool->mutex);
#else
    pthread_mutex_lock(&pool->mutex);
    pool->shutdown = 1;
    pthread_cond_broadcast(&pool->cond_nonempty);
    pthread_mutex_unlock(&pool->mutex);
    for (i = 0; i < pool->num_threads; i++)
        pthread_join(pool->threads[i], NULL);
    pthread_mutex_destroy(&pool->mutex);
    pthread_cond_destroy(&pool->cond_nonempty);
    pthread_cond_destroy(&pool->cond_nonfull);
#endif
    free(pool->queue);
    free(pool->threads);
    free(pool);
}

/*
 * package_cache.c — 배포 파일을 서버 기동 시 RAM에 미리 읽어 둠
 * 이유: 1000대 클라이언트가 동시에 받을 때 디스크 읽기 병목을 줄이기 위함
 */
#include "package_cache.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* packages/ 폴더 안의 파일 하나를 메모리로 읽기 */
static int load_one(cached_file_t *out, const char *dir, const char *fname) {
    char path[MAX_PATH_LEN];
    FILE *f;
    long sz;

    snprintf(path, sizeof(path), "%s/%s", dir, fname);
    strncpy(out->name, fname, MAX_PATH_LEN - 1);
    out->name[MAX_PATH_LEN - 1] = '\0';

    f = fopen(path, "rb");
    if (!f) return -1;
    fseek(f, 0, SEEK_END);
    sz = ftell(f);
    if (sz < 0) { fclose(f); return -1; }
    rewind(f);
    out->data = (char *)malloc((size_t)sz + 1);
    if (!out->data) { fclose(f); return -1; }
    if (fread(out->data, 1, (size_t)sz, f) != (size_t)sz) {
        free(out->data);
        out->data = NULL;
        fclose(f);
        return -1;
    }
    out->data[sz] = '\0';
    out->size = sz;
    fclose(f);
    return 0;
}

/* 설정에 적힌 update_file_*, migration_file_* 전부 캐시에 로드 */
int package_cache_load(package_cache_t *cache, const deploy_config_t *cfg) {
    int i;
    memset(cache, 0, sizeof(*cache));
    for (i = 0; i < cfg->update_file_count; i++) {
        if (load_one(&cache->update[cache->update_count], cfg->package_dir,
                     cfg->update_files[i]) == 0)
            cache->update_count++;
    }
    for (i = 0; i < cfg->migration_file_count; i++) {
        if (load_one(&cache->migration[cache->migration_count], cfg->package_dir,
                     cfg->migration_files[i]) == 0)
            cache->migration_count++;
    }
    cache->loaded = (cache->update_count > 0 || cache->migration_count > 0);
    return cache->loaded ? 0 : -1;
}

/* 서버 종료 시 malloc 한 메모리 해제 */
void package_cache_free(package_cache_t *cache) {
    int i;
    for (i = 0; i < cache->update_count; i++)
        free(cache->update[i].data);
    for (i = 0; i < cache->migration_count; i++)
        free(cache->migration[i].data);
    memset(cache, 0, sizeof(*cache));
}

const cached_file_t *package_cache_update(const package_cache_t *c, int i) {
    if (i < 0 || i >= c->update_count) return NULL;
    return &c->update[i];
}

const cached_file_t *package_cache_migration(const package_cache_t *c, int i) {
    if (i < 0 || i >= c->migration_count) return NULL;
    return &c->migration[i];
}

#!/bin/bash
# Ubuntu/Linux 동시 접속 부하 테스트
# 사용: ./load_test.sh [서버IP] [토큰] [횟수] [동시수]
set -e
HOST="${1:-127.0.0.1}"
TOKEN="${2:-UPD-DEMO-UPDATE-001}"
COUNT="${3:-30}"
PARALLEL="${4:-15}"
ROOT="$(cd "$(dirname "$0")/.." && pwd)"
CLIENT="$ROOT/deploy_client/deploy_client"
CONF="$ROOT/config/deploy_targets.conf"

if [ ! -x "$CLIENT" ]; then
  echo "먼저 프로젝트 루트에서: make"
  exit 1
fi

ok=0
fail=0
for i in $(seq 1 "$COUNT"); do
  while [ "$(jobs -rp | wc -l)" -ge "$PARALLEL" ]; do
    wait -n 2>/dev/null || wait
  done
  (
    if "$CLIENT" "$HOST" "$TOKEN" update "$CONF" >/dev/null 2>&1; then
      echo ok
    else
      echo fail
    fi
  ) &
done
wait
for r in $(jobs -p); do wait "$r" 2>/dev/null; done

echo "Load test done: total=$COUNT (see ok/fail above)"

# 동시 접속 부하 테스트 (기본 50, -Count 1000 으로 1000 연결 시뮬레이션)
param(
    [string]$TargetHost = "127.0.0.1",
    [string]$Token = "UPD-DEMO-UPDATE-001",
    [string]$Action = "update",
    [int]$Count = 50,
    [int]$MaxParallel = 32
)

$client = Join-Path $PSScriptRoot "..\deploy_client\deploy_client.exe"
$config = Join-Path $PSScriptRoot "..\config\deploy_targets.conf"
$ok = 0; $fail = 0

$jobs = @()
for ($i = 1; $i -le $Count; $i++) {
    while ((@(Get-Job -State Running)).Count -ge $MaxParallel) {
        Receive-Job -Job (Wait-Job -Any) | Out-Null
    }
    $jobs += Start-Job -ScriptBlock {
        param($c, $h, $t, $a, $cfg)
        & $c $h $t $a $cfg 2>&1 | Out-Null
        return $LASTEXITCODE
    } -ArgumentList $client, $TargetHost, $Token, $Action, $config
}

foreach ($j in $jobs) {
    $code = Receive-Job -Job $j -Wait
    if ($code -eq 0) { $ok++ } else { $fail++ }
    Remove-Job $j
}

Write-Host "Load test done: total=$Count ok=$ok fail=$fail"
