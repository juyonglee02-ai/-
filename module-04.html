/*
 * server.c — 배포 서버 (Ubuntu 에서 실행 권장)
 *
 * 하는 일:
 *   1) TCP 포트 listen (기본 9900)
 *   2) 클라이언트 접속마다 스레드 풀에 작업 전달
 *   3) 토큰 검증 후 패키지 파일 전송 (RAM 캐시 사용)
 *   4) NTP 동기화 요청 처리
 *
 * 실행 예:
 *   ./deploy_server ../config/deploy_targets.conf ../config/tokens.conf
 */
#include "../deploy_common/config.h"
#include "../deploy_common/thread_pool.h"
#include "../deploy_common/package_cache.h"
#include "../deploy_common/ntp_sync.h"
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#ifdef _WIN32
  #include <winsock2.h>
  #include <ws2tcpip.h>
  #pragma comment(lib, "ws2_32.lib")
  typedef int socklen_t;
  #define CLOSE_SOCKET closesocket
  #define sleep_sec(s) Sleep((s) * 1000)
#else
  #include <unistd.h>
  #include <pthread.h>
  #include <sys/socket.h>
  #include <netinet/in.h>
  #include <arpa/inet.h>
  #define SOCKET int
  #define INVALID_SOCKET (-1)
  #define SOCKET_ERROR (-1)
  #define CLOSE_SOCKET close

#endif

typedef struct {
    SOCKET sock;
} client_job_t;

static deploy_config_t g_cfg;              /* 설정 파일 내용 */
static token_entry_t g_tokens[MAX_TOKENS]; /* 허용 토큰 배열 */
static int g_token_count;
static package_cache_t g_cache;            /* RAM에 올려 둔 패키지 */
static thread_pool_t *g_pool;              /* 동시 처리용 스레드 풀 */
static volatile long g_active_clients;       /* 지금 처리 중인 연결 수 */
static volatile int g_shutdown;            /* 서버 종료 플래그 */

static int send_line(SOCKET sock, const char *msg) {
    char buf[MAX_LINE];
    snprintf(buf, sizeof(buf), "%s\n", msg);
    return send(sock, buf, (int)strlen(buf), 0) > 0 ? 0 : -1;
}

static int recv_line(SOCKET sock, char *buf, int size) {
    int n = 0;
    char c;
    while (n < size - 1) {
        if (recv(sock, &c, 1, 0) <= 0) return -1;
        if (c == '\n') break;
        if (c != '\r') buf[n++] = c;
    }
    buf[n] = '\0';
    return n;
}

/* 메모리 캐시에 있는 파일을 클라이언트로 전송 (디스크 읽기 없음) */
static int send_cached(SOCKET sock, const cached_file_t *cf) {
    char header[MAX_LINE];
    long sent = 0;
    int chunk;
    snprintf(header, sizeof(header), "FILE|%s|%ld", cf->name, cf->size);
    if (send_line(sock, header) != 0) return -1;
    while (sent < cf->size) {
        chunk = (int)((cf->size - sent) > 8192 ? 8192 : (cf->size - sent));
        if (send(sock, cf->data + sent, chunk, 0) <= 0) return -1;
        sent += chunk;
    }
    return send_line(sock, "FILE_END");
}

static long file_size_path(const char *path) {
    FILE *f = fopen(path, "rb");
    long sz;
    if (!f) return -1;
    fseek(f, 0, SEEK_END);
    sz = ftell(f);
    fclose(f);
    return sz;
}

static int send_file_disk(SOCKET sock, const char *filepath, const char *name) {
    FILE *f;
    char buf[8192];
    size_t rd;
    long sz;
    char header[MAX_LINE];
    long sent = 0;

    sz = file_size_path(filepath);
    if (sz < 0) return -1;
    snprintf(header, sizeof(header), "FILE|%s|%ld", name, sz);
    if (send_line(sock, header) != 0) return -1;
    f = fopen(filepath, "rb");
    if (!f) return -1;
    while (sent < sz && (rd = fread(buf, 1, sizeof(buf), f)) > 0) {
        if (send(sock, buf, (int)rd, 0) <= 0) {
            fclose(f);
            return -1;
        }
        sent += (long)rd;
    }
    fclose(f);
    return send_line(sock, "FILE_END");
}

static deploy_action_t parse_action(const char *s) {
    if (strcmp(s, "update") == 0) return ACTION_UPDATE;
    if (strcmp(s, "migration") == 0) return ACTION_MIGRATION;
    if (strcmp(s, "ntp") == 0) return ACTION_NTP;
    return ACTION_UNKNOWN;
}

static void handle_ntp(SOCKET sock) {
    char detail[MAX_LINE];
    if (ntp_sync_system(g_cfg.ntp_server, detail, sizeof(detail)) == 0) {
        char msg[MAX_LINE];
        snprintf(msg, sizeof(msg), "NTP|%s|%s", g_cfg.ntp_server, detail);
        send_line(sock, "OK|ntp started");
        send_line(sock, msg);
        send_line(sock, "OK|ntp complete");
    } else {
        send_line(sock, "ERR|ntp sync failed");
    }
}

static void send_packages(SOCKET sock, deploy_action_t action) {
    int i, n;
    const cached_file_t *cf;
    char filepath[MAX_PATH_LEN];

    if (action == ACTION_UPDATE) n = g_cfg.update_file_count;
    else n = g_cfg.migration_file_count;

    for (i = 0; i < n; i++) {
        if (g_cache.loaded && g_cfg.package_cache_enabled) {
            cf = (action == ACTION_UPDATE)
                     ? package_cache_update(&g_cache, i)
                     : package_cache_migration(&g_cache, i);
            if (cf && send_cached(sock, cf) != 0) {
                send_line(sock, "ERR|cache send failed");
                return;
            }
        } else {
            if (action == ACTION_UPDATE)
                snprintf(filepath, sizeof(filepath), "%s/%s",
                         g_cfg.package_dir, g_cfg.update_files[i]);
            else
                snprintf(filepath, sizeof(filepath), "%s/%s",
                         g_cfg.package_dir, g_cfg.migration_files[i]);
            if (action == ACTION_UPDATE)
                send_file_disk(sock, filepath, g_cfg.update_files[i]);
            else
                send_file_disk(sock, filepath, g_cfg.migration_files[i]);
        }
    }
}

/* 한 클라이언트 연결 처리: 토큰|action 받고 파일 전송 또는 NTP */
static void handle_client_socket(SOCKET client_sock) {
    char line[MAX_LINE];
    char token[MAX_TOKEN_LEN];
    char action_str[32];
    char *sep;
    deploy_action_t action;

    if (recv_line(client_sock, line, sizeof(line)) < 0) {
        send_line(client_sock, "ERR|read failed");
        return;
    }

    strncpy(token, line, MAX_TOKEN_LEN - 1);
    token[MAX_TOKEN_LEN - 1] = '\0';
    sep = strchr(token, '|');
    if (!sep) {
        send_line(client_sock, "ERR|format: TOKEN|action");
        return;
    }
    *sep = '\0';
    strncpy(action_str, sep + 1, sizeof(action_str) - 1);
    action = parse_action(action_str);
    if (action == ACTION_UNKNOWN) {
        send_line(client_sock, "ERR|unknown action");
        return;
    }

    if (!validate_token(token, action, g_tokens, g_token_count)) {
        send_line(client_sock, "ERR|invalid or expired token");
        return;
    }

    if (action == ACTION_NTP) {
        handle_ntp(client_sock);
        return;
    }

    if (action == ACTION_UPDATE) {
        send_line(client_sock, "OK|update started");
        send_packages(client_sock, ACTION_UPDATE);
        send_line(client_sock, "OK|update complete");
    } else {
        send_line(client_sock, "OK|migration started");
        send_packages(client_sock, ACTION_MIGRATION);
        send_line(client_sock, "OK|migration complete");
    }
}

/* 스레드 풀에서 실행되는 함수 — 실제 배포 작업 */
static void client_worker(void *arg) {
    client_job_t *job = (client_job_t *)arg;
    handle_client_socket(job->sock);
    CLOSE_SOCKET(job->sock);
    free(job);
#ifdef _WIN32
    InterlockedDecrement((volatile LONG *)&g_active_clients);
#else
    __sync_fetch_and_sub(&g_active_clients, 1);
#endif
}

#ifdef _WIN32
static DWORD WINAPI ntp_background_thread(LPVOID unused) {
    char detail[MAX_LINE];
    (void)unused;
    while (!g_shutdown) {
        int mins = g_cfg.ntp_sync_interval_min > 0 ? g_cfg.ntp_sync_interval_min : 30;
        sleep_sec(mins * 60);
        if (g_shutdown) break;
        ntp_sync_system(g_cfg.ntp_server, detail, sizeof(detail));
        printf("[NTP] periodic sync: %s\n", detail);
    }
    return 0;
}
#else
static void *ntp_background_thread(void *unused) {
    char detail[MAX_LINE];
    (void)unused;
    while (!g_shutdown) {
        int mins = g_cfg.ntp_sync_interval_min > 0 ? g_cfg.ntp_sync_interval_min : 30;
        sleep_sec(mins * 60);
        if (g_shutdown) break;
        ntp_sync_system(g_cfg.ntp_server, detail, sizeof(detail));
        printf("[NTP] periodic sync: %s\n", detail);
    }
    return NULL;
}
#endif

/* 서버 시작: 설정 로드 → 캐시 로드 → NTP → listen 루프 */
int main(int argc, char *argv[]) {
    const char *cfg_path = "../config/deploy_targets.conf";
    const char *tok_path = "../config/tokens.conf";
    SOCKET listen_sock, client_sock;
    struct sockaddr_in addr, client_addr;
    socklen_t client_len;
    char ntp_detail[MAX_LINE];

#ifdef _WIN32
    WSADATA wsa;
#endif
#ifndef _WIN32
    pthread_t ntp_thread;
#endif

#ifdef _WIN32
    if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0) return 1;
#endif

    if (argc > 1) cfg_path = argv[1];
    if (argc > 2) tok_path = argv[2];

    token_lock_init();
    if (load_deploy_config(cfg_path, &g_cfg) != 0) {
        fprintf(stderr, "Cannot load config: %s\n", cfg_path);
        return 1;
    }
    g_token_count = load_tokens(tok_path, g_tokens, MAX_TOKENS);

    if (g_cfg.package_cache_enabled) {
        if (package_cache_load(&g_cache, &g_cfg) == 0)
            printf("Package cache: update=%d migration=%d files in RAM\n",
                   g_cache.update_count, g_cache.migration_count);
        else
            printf("Warning: package cache load failed, using disk I/O\n");
    }

    if (g_cfg.ntp_sync_on_startup) {
        if (ntp_sync_system(g_cfg.ntp_server, ntp_detail, sizeof(ntp_detail)) == 0)
            printf("[NTP] startup sync OK: %s\n", ntp_detail);
        else
            printf("[NTP] startup sync skipped/failed: %s (run as admin on Windows)\n", ntp_detail);
    }

#ifdef _WIN32
    CreateThread(NULL, 0, ntp_background_thread, NULL, 0, NULL);
#else
    pthread_create(&ntp_thread, NULL, ntp_background_thread, NULL);
#endif

    g_pool = thread_pool_create(g_cfg.worker_threads, g_cfg.max_queue_size);
    if (!g_pool) {
        fprintf(stderr, "thread pool create failed\n");
        return 1;
    }

    printf("Deploy server (high-concurrency)\n");
    printf("  port=%d workers=%d backlog=%d max_concurrent=%d tokens=%d\n",
           g_cfg.server_port, g_cfg.worker_threads, g_cfg.listen_backlog,
           g_cfg.max_concurrent, g_token_count);

    listen_sock = socket(AF_INET, SOCK_STREAM, 0);
    if (listen_sock == INVALID_SOCKET) return 1;

#ifndef _WIN32
    { int opt = 1; setsockopt(listen_sock, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt)); }
#endif

    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons((unsigned short)g_cfg.server_port);

    if (bind(listen_sock, (struct sockaddr *)&addr, sizeof(addr)) == SOCKET_ERROR ||
        listen(listen_sock, g_cfg.listen_backlog) == SOCKET_ERROR) {
        fprintf(stderr, "bind/listen failed\n");
        return 1;
    }

    printf("Listening ... (ready for 1000+ clients)\n");
    while (1) {
        client_len = sizeof(client_addr);
        client_sock = accept(listen_sock, (struct sockaddr *)&client_addr, &client_len);
        if (client_sock == INVALID_SOCKET) continue;

        if (g_active_clients >= g_cfg.max_concurrent) {
            send_line(client_sock, "ERR|server busy, retry later");
            CLOSE_SOCKET(client_sock);
            continue;
        }

        if (thread_pool_queue_size(g_pool) >= g_cfg.max_queue_size) {
            send_line(client_sock, "ERR|queue full, retry later");
            CLOSE_SOCKET(client_sock);
            continue;
        }

        {
            client_job_t *job = (client_job_t *)malloc(sizeof(client_job_t));
            if (!job) {
                CLOSE_SOCKET(client_sock);
                continue;
            }
            job->sock = client_sock;
#ifdef _WIN32
            InterlockedIncrement((volatile LONG *)&g_active_clients);
#else
            __sync_fetch_and_add(&g_active_clients, 1);
#endif
            if (thread_pool_submit(g_pool, client_worker, job) != 0) {
#ifdef _WIN32
                InterlockedDecrement((volatile LONG *)&g_active_clients);
#else
                __sync_fetch_and_sub(&g_active_clients, 1);
#endif
                send_line(client_sock, "ERR|submit failed");
                CLOSE_SOCKET(client_sock);
                free(job);
            }
        }
    }

    g_shutdown = 1;
    package_cache_free(&g_cache);
    thread_pool_destroy(g_pool);
    token_lock_shutdown();
    CLOSE_SOCKET(listen_sock);
#ifdef _WIN32
    WSACleanup();
#endif
    return 0;
}

  #define MKDIR(p) mkdir(p, 0755)
#endif

static deploy_config_t g_cfg;  /* deploy_targets.conf 에서 읽은 설정 */

/* 프로토콜: 한 줄 텍스트 보내기 (끝에 \n 붙임) */
static int send_line(SOCKET sock, const char *msg) {
    char buf[MAX_LINE];
    snprintf(buf, sizeof(buf), "%s\n", msg);
    return send(sock, buf, (int)strlen(buf), 0) > 0 ? 0 : -1;
}

static int recv_line(SOCKET sock, char *buf, int size) {
    int n = 0;
    char c;
    while (n < size - 1) {
        if (recv(sock, &c, 1, 0) <= 0) return -1;
        if (c == '\n') break;
        if (c != '\r') buf[n++] = c;
    }
    buf[n] = '\0';
    return n;
}

static void ensure_dir(const char *path) {
    char tmp[MAX_PATH_LEN];
    char *p;
    strncpy(tmp, path, MAX_PATH_LEN - 1);
    for (p = tmp + 1; *p; p++) {
        if (*p == '/' || *p == '\\') {
            *p = '\0';
            MKDIR(tmp);
            *p = '/';
        }
    }
    MKDIR(tmp);
}

/* TCP는 한 번에 원하는 만큼 안 올 수 있어서, 정확히 need 바이트만큼 받을 때까지 반복 */
static int recv_exact(SOCKET sock, char *buf, int need) {
    int got = 0, n;
    while (got < need) {
        n = recv(sock, buf + got, need - got, 0);
        if (n <= 0) return -1;
        got += n;
    }
    return got;
}

/*
 * 서버가 보낸 FILE|이름|크기 메시지 다음의 바이너리를 파일로 저장
 */
static int recv_file(SOCKET sock, const char *install_dir, const char *line) {
    char filename[MAX_PATH_LEN];
    long expected = 0, received = 0;
    char outpath[MAX_PATH_LEN];
    FILE *f;
    char buf[4096];
    int n, chunk;
    char endline[MAX_LINE];

    if (sscanf(line, "FILE|%259[^|]|%ld", filename, &expected) < 2)
        return -1;

    snprintf(outpath, sizeof(outpath), "%s/%s", install_dir, filename);
    ensure_dir(install_dir);
    f = fopen(outpath, "wb");
    if (!f) {
        fprintf(stderr, "Cannot write %s\n", outpath);
        return -1;
    }
    printf("Receiving -> %s (%ld bytes)\n", outpath, expected);

    while (received < expected) {
        chunk = (int)((expected - received) < (long)sizeof(buf) ? (expected - received) : (long)sizeof(buf));
        n = recv_exact(sock, buf, chunk);
        if (n <= 0) {
            fclose(f);
            return -1;
        }
        fwrite(buf, 1, (size_t)n, f);
        received += n;
    }
    fclose(f);

    if (recv_line(sock, endline, sizeof(endline)) < 0 || strcmp(endline, "FILE_END") != 0) {
        fprintf(stderr, "Expected FILE_END, got: %s\n", endline);
        return -1;
    }
    printf("Saved %s\n", outpath);
    return 0;
}

static void report_flask(const char *token, const char *status, const char *message) {
    char cmd[MAX_LINE * 2];
    if (g_cfg.flask_report_url[0] == '\0') return;
#ifdef _WIN32
    snprintf(cmd, sizeof(cmd),
        "powershell -Command \"Invoke-RestMethod -Uri '%s' -Method POST "
        "-ContentType 'application/json' -Body '{\\\"token\\\":\\\"%s\\\","
        "\\\"status\\\":\\\"%s\\\",\\\"message\\\":\\\"%s\\\"}'\" 2>nul",
        g_cfg.flask_report_url, token, status, message);
#else
    snprintf(cmd, sizeof(cmd),
        "curl -s -X POST -H 'Content-Type: application/json' "
        "-d '{\"token\":\"%s\",\"status\":\"%s\",\"message\":\"%s\"}' '%s' 2>/dev/null",
        token, status, message, g_cfg.flask_report_url);
#endif
    system(cmd);
}

/*
 * main — 프로그램 시작점
 * argv[1]=서버IP, argv[2]=토큰, argv[3]=update|migration|ntp
 */
int main(int argc, char *argv[]) {
    const char *host = "127.0.0.1";
    const char *token;
    const char *action_str;
    const char *cfg_path = "../config/deploy_targets.conf";
    SOCKET sock;
    struct sockaddr_in addr;
    char req[MAX_LINE];
    char line[MAX_LINE];
    int port;

#ifdef _WIN32
    WSADATA wsa;
#endif

    if (argc < 4) {
        fprintf(stderr, "Usage: %s <host> <token> <update|migration|ntp> [config]\n", argv[0]);
        return 1;
    }
    host = argv[1];
    token = argv[2];
    action_str = argv[3];
    if (argc > 4) cfg_path = argv[4];

    if (load_deploy_config(cfg_path, &g_cfg) != 0) {
        fprintf(stderr, "Cannot load config\n");
        return 1;
    }
    port = g_cfg.server_port;

#ifdef _WIN32
    WSAStartup(MAKEWORD(2, 2), &wsa);
#endif

    sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock == INVALID_SOCKET) return 1;

    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_port = htons((unsigned short)port);
#ifdef _WIN32
    addr.sin_addr.s_addr = inet_addr(host);
    if (addr.sin_addr.s_addr == INADDR_NONE) {
        fprintf(stderr, "Invalid host\n");
        return 1;
    }
#else
    if (inet_pton(AF_INET, host, &addr.sin_addr) <= 0) {
        fprintf(stderr, "Invalid host\n");
        return 1;
    }
#endif

    printf("Connecting to %s:%d ...\n", host, port);
    if (connect(sock, (struct sockaddr *)&addr, sizeof(addr)) == SOCKET_ERROR) {
        fprintf(stderr, "Connect failed\n");
        report_flask(token, "failed", "connect failed");
        return 1;
    }

    snprintf(req, sizeof(req), "%s|%s", token, action_str);
    send_line(sock, req);
    printf("Sent: %s\n", req);

    while (recv_line(sock, line, sizeof(line)) > 0) {
        printf("< %s\n", line);
        if (strncmp(line, "ERR|", 4) == 0) {
            report_flask(token, "failed", line + 4);
            CLOSE_SOCKET(sock);
            return 1;
        }
        if (strncmp(line, "FILE|", 5) == 0) {
            if (recv_file(sock, g_cfg.client_install_dir, line) != 0) {
                report_flask(token, "failed", "file receive error");
                CLOSE_SOCKET(sock);
                return 1;
            }
            continue;
        }
        if (strncmp(line, "NTP|", 4) == 0) {
            char server[MAX_PATH_LEN], detail[MAX_LINE];
            if (sscanf(line, "NTP|%259[^|]|%511[^\n]", server, detail) >= 1) {
                char msg[MAX_LINE];
                printf("Applying NTP peer: %s\n", server);
                if (g_cfg.ntp_sync_clients) {
                    if (ntp_sync_system(server, msg, sizeof(msg)) == 0)
                        printf("Client NTP sync OK: %s\n", msg);
                    else
                        printf("Client NTP sync failed: %s\n", msg);
                }
            }
            continue;
        }
        if (strcmp(line, "OK|update complete") == 0 ||
            strcmp(line, "OK|migration complete") == 0 ||
            strcmp(line, "OK|ntp complete") == 0) {
            report_flask(token, "success", line + 3);
            break;
        }
        if (strncmp(line, "OK|", 3) == 0)
            continue;
    }

    CLOSE_SOCKET(sock);
#ifdef _WIN32
    WSACleanup();
#endif
    printf("Done.\n");
    return 0;
}
