models.py — Flask 가 SQLite DB에 저장하는 테이블 정의 (ORM)

"""
from datetime import datetime, timezone

from flask_sqlalchemy import SQLAlchemy

db = SQLAlchemy()


def utcnow():
    return datetime.now(timezone.utc)


class TokenFormat(db.Model):
    """GUI에서 정의하는 토큰 사용 형식"""

    __tablename__ = "token_formats"

    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(64), nullable=False, unique=True)
    token_type = db.Column(db.String(32), nullable=False)  # update, migration
    prefix = db.Column(db.String(32), default="")
    pattern = db.Column(db.String(256), default="")  # 예: UPDATE-{client}-{version}
    description = db.Column(db.Text, default="")
    enabled = db.Column(db.Boolean, default=True)
    created_at = db.Column(db.DateTime, default=utcnow)

    tokens = db.relationship("DeployToken", backref="format", lazy=True)


class DeployToken(db.Model):
    """발급된 배포 토큰"""

    __tablename__ = "deploy_tokens"

    id = db.Column(db.Integer, primary_key=True)
    token_value = db.Column(db.String(128), nullable=False, unique=True)
    format_id = db.Column(db.Integer, db.ForeignKey("token_formats.id"), nullable=False)
    client_id = db.Column(db.Integer, db.ForeignKey("clients.id"))
    expires_at = db.Column(db.DateTime)
    used_count = db.Column(db.Integer, default=0)
    max_uses = db.Column(db.Integer, default=1)
    active = db.Column(db.Boolean, default=True)
    created_at = db.Column(db.DateTime, default=utcnow)


class Client(db.Model):
    """배포 대상 클라이언트"""

    __tablename__ = "clients"

    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(64), nullable=False, unique=True)
    host = db.Column(db.String(128), default="127.0.0.1")
    port = db.Column(db.Integer, default=9900)
    version = db.Column(db.String(32), default="1.0.0")
    status = db.Column(db.String(32), default="idle")  # idle, updating, error, ok
    last_seen = db.Column(db.DateTime)
    created_at = db.Column(db.DateTime, default=utcnow)

    deployments = db.relationship("Deployment", backref="client", lazy=True)
    tokens = db.relationship("DeployToken", backref="client", lazy=True)


class NtpSyncLog(db.Model):
    """서버/클라이언트 NTP 동기화 이력"""

    __tablename__ = "ntp_sync_logs"

    id = db.Column(db.Integer, primary_key=True)
    source = db.Column(db.String(32), default="server")  # server, client
    ntp_server = db.Column(db.String(128), default="")
    status = db.Column(db.String(32), default="success")
    detail = db.Column(db.Text, default="")
    synced_at = db.Column(db.DateTime, default=utcnow)


class Deployment(db.Model):
    """배포/업데이트 이력"""

    __tablename__ = "deployments"

    id = db.Column(db.Integer, primary_key=True)
    client_id = db.Column(db.Integer, db.ForeignKey("clients.id"), nullable=False)
    action = db.Column(db.String(32), nullable=False)  # update, migration
    token_used = db.Column(db.String(128))
    version_from = db.Column(db.String(32))
    version_to = db.Column(db.String(32))
    status = db.Column(db.String(32), default="pending")  # pending, running, success, failed
    message = db.Column(db.Text, default="")
    started_at = db.Column(db.DateTime, default=utcnow)
    finished_at = db.Column(db.DateTime)

def generate_token_value(fmt: TokenFormat, client: Client | None = None) -> str:
    random_part = secrets.token_hex(4).upper()
    version = client.version if client else "0.0.0"
    client_id = str(client.id) if client else "0"
    pattern = fmt.pattern or "{prefix}-{random}"
    value = (
        pattern.replace("{prefix}", fmt.prefix or "")
        .replace("{client_id}", client_id)
        .replace("{version}", version)
        .replace("{random}", random_part)
    )
    return value.strip("-")

def issue_token():
    fmt = TokenFormat.query.get_or_404(int(request.form["format_id"]))
    client = None
    if request.form.get("client_id"):
        client = Client.query.get(int(request.form["client_id"]))

    token = DeployToken(

        client_id=client.id if client else None,

        max_uses=int(request.form.get("max_uses", 1)),
    )
    db.session.add(token)
    db.session.commit()
    flash(f"토큰 발급: {value}", "success")
    return redirect(url_for("tokens"))

    token = DeployToken(
        token_value=token_value,
        format_id=fmt.id,
        client_id=client.id,
        expires_at=utcnow() + timedelta(hours=24),
        max_uses=1,
    )
    db.session.add(token)

    deployment = Deployment(
        client_id=client.id,
        action=action,
        token_used=token_value,
        version_from=client.version,
        version_to=request.form.get("version_to", client.version),
        status="pending",
        message="배포 대기 중 (C 클라이언트에서 토큰으로 실행)",
    )
    client.status = "updating"
    db.session.add(deployment)
    db.session.commit()

@app.route("/api/stats/deployments")
def api_stats_deployments():
    """업데이트 현황 통계 (일별)"""
    rows = (
        db.session.query(
            func.date(Deployment.started_at).label("day"),
            Deployment.status,
            func.count(Deployment.id),
        )
        .group_by(func.date(Deployment.started_at), Deployment.status)
        .all()
    )
    days = sorted({str(r.day) for r in rows if r.day})
    statuses = ["success", "failed", "pending", "running"]
    datasets = {s: [0] * len(days) for s in statuses}
    for r in rows:
        if r.day:
            idx = days.index(str(r.day))
            if r.status in datasets:
                datasets[r.status][idx] = r[2]
    return jsonify({"labels": days, "datasets": datasets})

def _read_deploy_conf():
    """deploy_targets.conf 키=값 파싱"""
    conf = {}
    path = app.config.get("DEPLOY_TARGETS_CONFIG")
    if not path or not os.path.isfile(path):
        return conf
    with open(path, encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            conf[k.strip()] = v.strip()
    return conf

def api_report():
    """C 클라이언트/서버에서 배포 결과 보고"""
    data = request.get_json(force=True, silent=True) or {}
    token_value = data.get("token", "")
    status = data.get("status", "success")
    message = data.get("message", "")
    version = data.get("version", "")

    if "ntp" in message.lower() or data.get("action") == "ntp":
        db.session.add(
            NtpSyncLog(
                source="client",
                ntp_server=data.get("ntp_server", ""),
                status=status,
                detail=message,
            )
        )

    token = DeployToken.query.filter_by(token_value=token_value, active=True).first()
    if token:
        token.used_count += 1
        if token.used_count >= token.max_uses:
            token.active = False

    deployment = (
        Deployment.query.filter_by(token_used=token_value, status="pending")
        .order_by(Deployment.started_at.desc())
        .first()
    )
    if deployment:
        deployment.status = status
        deployment.message = message
        deployment.finished_at = utcnow()
        if version:
            deployment.version_to = version
        client = Client.query.get(deployment.client_id)
        if client:
            client.status = "ok" if status == "success" else "error"
            if version:
                client.version = version
            client.last_seen = utcnow()

    db.session.commit()
    return jsonify({"ok": True})
